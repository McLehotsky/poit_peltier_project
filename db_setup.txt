CREATE DATABASE IF NOT EXISTS ml_poit;

USE ml_poit;

CREATE TABLE IF NOT EXISTS measurements (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    temperature FLOAT,
    pump_pwm    INT,
    tec_pwm     INT,
    setpoint    FLOAT,
    timestamp   DATETIME
);
